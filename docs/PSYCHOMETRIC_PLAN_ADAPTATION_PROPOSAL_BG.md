  const dimensions = ['E', 'C', 'O', 'A', 'N', 'I', 'R'];
  const dimensionScores = {};
  
  for (const dim of dimensions) {
    const correlations = DIMENSION_CORRELATIONS[dim];
    let points = 0;
    let maxPoints = 0;
    
    for (const corr of correlations) {
      const answer = initialAnswers[corr.questionId];
      const psychScore = psychProfile.dimensions[dim];
      
      if (matchesExpectedPattern(answer, psychScore, corr.pattern)) {
        points += corr.weight;
      }
      maxPoints += corr.weight;
    }
    
    dimensionScores[dim] = points / maxPoints;
  }
  
  // Weighted average –ø–æ —Ç–µ–≥–ª–æ –Ω–∞ –∏–∑–º–µ—Ä–µ–Ω–∏—è—Ç–∞
  const weights = { E: 0.15, C: 0.25, O: 0.10, A: 0.10, N: 0.20, I: 0.15, R: 0.05 };
  let weightedSum = 0;
  
  for (const [dim, score] of Object.entries(dimensionScores)) {
    weightedSum += score * weights[dim];
  }
  
  return weightedSum;
}
